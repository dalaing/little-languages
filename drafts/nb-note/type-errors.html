Tokens and spans

trimTrailingWhitespace :: Span
                       -> B.ByteString
                       -> Span
trimTrailingWhitespace (Span s e l) bs =
    Span s e' l
  where
    e' = s <> delta (B.take n bs)
    n =
      B.length bs - w
    w =
      B.length .
      C.takeWhile isSpace .
      B.reverse $
      bs
withTokenSpan :: DeltaParsing m
              => m a
              -> m (a, Span)
withTokenSpan p = do
  (a :~ span, bs) <- slicedWith (,) . spanned $ p
  return (a, trimTrailingWhitespace span bs)
withSpan :: DeltaParsing m
         => m (Term Span)
         -> m (Term Span)
withSpan p = do
  (tm, sp) <- withTokenSpan p
  return $ TmNote sp tm

From the terms to the types

inferTmNote :: MonadError (TypeError Span) m
            => (Term Span -> m (Type Span))
            -> Term Span
            -> Maybe (m (Type Span))
inferTmNote infer (TmNote n tm) = Just $ do
  ty <- infer tm
  return $ TyNote n ty
inferTmNote _ _ =
  Nothing
